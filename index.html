import can
import time

class TorqueRequestTest:
    def __init__(self):
        self.bus = can.interface.Bus(channel=0, bustype='pcan')

    def test_torque_normal(self):
        """扭矩请求-正向正常值测试"""
        msg = can.Message(arbitration_id=0x100, data=[0x00, 0x50, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00])
        self.bus.send(msg)
        time.sleep(0.1)
        response = self.wait_for_response(timeout=1.0)
        assert response.data[0] == 0x01, "扭矩响应错误"
